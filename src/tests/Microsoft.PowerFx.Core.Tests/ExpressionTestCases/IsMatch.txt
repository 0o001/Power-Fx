>> IsMatch("Hi", "H")

>> IsMatch("Hi", "Hi")
true

>> IsMatch("Hi", "Hi", MatchOptions.Complete)

>> IsMatch("Bye", "H")

>> IsMatch("Hi", "H", "c")

>> IsMatch("Hi", "H", MatchOptions.Contains)
true

>> IsMatch("Hi", "H", MatchOptions.Contains & MatchOptions.Complete)
false

>> IsMatch("Hi", "H", MatchOptions.Contains & MatchOptions.BeginWith)
true

>> IsMatch("Hi", "H", "$c")

>> IsMatch("Hi", "H", "c$")

>> IsMatch("Hi", "H", "^c")
true

>> IsMatch("Hi", "H", "c^")
true

>> IsMatch("Hi", "i", "c$")

>> IsMatch("Hi", "I", "c$")

>> IsMatch("Hi", "I", "c$i")

>> IsMatch("Hi", "H.*")

>> IsMatch("Hi", "H.*i")
